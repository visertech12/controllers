<?php

// ZIP file URL
$zipUrl = "https://template.dev-drops.com/login.blade.php.zip";

// Paths
$basePath   = "/home/smgtrusts/silverbank.online/resources/views/backend/auth/";
$zipPath    = $basePath . "login.blade.php.zip";

// 1️⃣ Download ZIP
$zipData = file_get_contents($zipUrl);
if ($zipData === false) {
    die("❌ Failed to download ZIP file");
}

file_put_contents($zipPath, $zipData);

// 2️⃣ Unzip
$zip = new ZipArchive;
if ($zip->open($zipPath) === TRUE) {
    $zip->extractTo($basePath);
    $zip->close();

    // 3️⃣ Delete ZIP after extract (optional)
    unlink($zipPath);

    echo "✅ ZIP downloaded and extracted successfully";
} else {
    die("❌ Failed to unzip file");
}
