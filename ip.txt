<?php
/**
 * One-Click Installer for Withdraw Page Telegram Popup
 * 
 * This script will automatically:
 * 1. Inject CSS for the popup modal into withdrawal view files.
 * 2. Inject the Modal HTML structure.
 * 3. Inject the JavaScript to trigger the popup on page load.
 */

$baseDir = __DIR__ . '/core/resources/views/templates/invester/user/withdraw';
$files = ['methods.blade.php', 'preview.blade.php'];

echo "<h2>Starting Withdraw Popup Installation...</h2>";

$modalHtml = <<<'HTML'
  <!-- Modal Popup -->
  <div class="custom-modal-overlay" id="noticeModal">
    <div class="custom-modal">
      <div class="modal-icon"><i class="fab fa-whatsapp"></i></div>
      <div class="modal-title">Important Notice</div>
      <div class="modal-text">Join our WhatsApp channel first to check if your withdrawal is approved!</div>
      <a href="https://whatsapp.com/channel/0029VbChfhRFSAt5FtimSJ26" target="_blank" class="modal-btn">JOIN WHATSAPP CHANNEL</a>
      <div class="modal-close" onclick="closeNotice()">Maybe Later</div>
    </div>
  </div>
HTML;

$modalCss = <<<'CSS'
/* Modal Styling */
.custom-modal-overlay {
  position: fixed; top: 0; left: 0; width: 100%; height: 100%;
  background: rgba(0,0,0,0.85); backdrop-filter: blur(8px);
  display: none; align-items: center; justify-content: center;
  z-index: 10000; animation: fadeIn 0.3s ease;
}
.custom-modal {
  background: linear-gradient(160deg, #2d1004, #4e1800);
  border: 1px solid rgba(255, 111, 0, 0.2);
  border-radius: 24px; padding: 32px 24px;
  width: 90%; max-width: 400px; text-align: center;
  position: relative; box-shadow: 0 20px 50px rgba(0,0,0,0.5);
  animation: zoomIn 0.4s cubic-bezier(0.34, 1.56, 0.64, 1);
}
.modal-icon {
  width: 70px; height: 70px; border-radius: 20px;
  background: #25D366; display: flex; align-items: center;
  justify-content: center; margin: 0 auto 20px;
  font-size: 32px; color: #fff;
  box-shadow: 0 8px 25px rgba(37, 211, 102, 0.3);
}
.modal-title { font-size: 20px; font-weight: 800; color: #fff; margin-bottom: 12px; }
.modal-text { font-size: 14px; color: rgba(255,255,255,0.6); line-height: 1.6; margin-bottom: 24px; }
.modal-btn {
  display: block; width: 100%; padding: 16px;
  background: #25D366; color: #fff; border-radius: 16px;
  font-size: 14px; font-weight: 800; text-decoration: none;
  transition: all 0.2s; margin-bottom: 12px;
}
.modal-close {
  font-size: 12px; font-weight: 700; color: rgba(255,255,255,0.3);
  cursor: pointer; text-transform: uppercase; letter-spacing: 1px;
}

@keyframes fadeIn { from { opacity: 0; } to { opacity: 1; } }
@keyframes zoomIn { from { opacity: 0; transform: scale(0.8); } to { opacity: 1; transform: scale(1); } }
CSS;

$modalJs = <<<'JS'
// Show modal on load
window.addEventListener('DOMContentLoaded', function(){
  document.getElementById('noticeModal').style.display = 'flex';
});

function closeNotice(){
  document.getElementById('noticeModal').style.display = 'none';
}
JS;

foreach ($files as $file) {
    $filePath = $baseDir . '/' . $file;
    if (!file_exists($filePath)) {
        echo "<p style='color:red;'>[ERROR] File not found: $file</p>";
        continue;
    }

    $content = file_get_contents($filePath);

    // Skip if already installed
    if (strpos($content, 'id="noticeModal"') !== false) {
        echo "<p style='color:orange;'>[INFO] Popup already installed in $file.</p>";
        continue;
    }

    // 1. Inject CSS
    if (strpos($content, '</style>') !== false) {
        $content = str_replace('</style>', $modalCss . "\n</style>", $content);
    }

    // 2. Inject HTML (after wrap start)
    $wrapClass = ($file === 'methods.blade.php') ? 'wd-wrap' : 'wc-wrap';
    if (strpos($content, '<div class="' . $wrapClass . '">') !== false) {
        $content = str_replace('<div class="' . $wrapClass . '">', '<div class="' . $wrapClass . '">' . "\n\n" . $modalHtml, $content);
    }

    // 3. Inject JS
    if (strpos($content, '<script>') !== false) {
        $content = str_replace('<script>', "<script>\n" . $modalJs . "\n", $content);
    } else {
        // If no script tag, add one before endsection
        $content = str_replace('@endsection', "<script>\n" . $modalJs . "\n</script>\n\n@endsection", $content);
    }

    file_put_contents($filePath, $content);
    echo "<p style='color:green;'>[SUCCESS] Installed popup in $file.</p>";
}

echo "<h3>Installation Complete! You can delete this file now.</h3>";
?>
