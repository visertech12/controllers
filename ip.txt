<?php
/**
 * PHP Script to download and overwrite plan.blade.php
 */

$targetFile = '/home/azizihur/public_html/core/resources/views/templates/invester/partials/plan.blade.php';
$url = 'http://txt.xforo.co/plan.txt';

echo "<h2>Fetching plan.blade.php from URL...</h2>";

// Fetch the content from the URL
$content = file_get_contents($url);

if ($content === false) {
    die("<p style='color:red;'>Error: Failed to fetch the content from $url</p>");
}

if (empty(trim($content))) {
    die("<p style='color:red;'>Error: The fetched content is empty! Make sure the URL is accessible.</p>");
}

// Write the content to the target file
if (file_put_contents($targetFile, $content) !== false) {
    echo "<p style='color:green;'><b>Success:</b> The file at $targetFile has been successfully updated with the code from the link!</p>";
} else {
    echo "<p style='color:red;'><b>Error:</b> Failed to write to $targetFile. Please check file permissions on your server.</p>";
}

echo "<h3>Update Complete! You can now delete this script.</h3>";
?>
