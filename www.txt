<?php

final class TradeVexaExternalSecurityVerifier
{
    private const PUBLIC_KEY_SHA256 = '8155fb02e25ed150f36d4e277bdead783985f8525d15a6a5fc0dfd9889e5eaaf';
    private const MAX_PAYLOAD_BYTES = 1048576;

    private string $securityRoot;
    private string $publicRoot;

    public function __construct(?string $securityRoot = null, ?string $publicRoot = null)
    {
        $this->securityRoot = rtrim($securityRoot ?: __DIR__, DIRECTORY_SEPARATOR);
        $this->publicRoot = rtrim($publicRoot ?: dirname(__DIR__) . '/public_html', DIRECTORY_SEPARATOR);
    }

    public function verifyAll(bool $ignoreEmergencyLock = false): array
    {
        if (!$ignoreEmergencyLock && is_file($this->securityRoot . '/DEPOSITS_LOCKED')) {
            throw new RuntimeException('Emergency deposit lock is active.');
        }

        $externalManifest = $this->verifiedPayload(
            $this->securityRoot . '/external_security_files.sha256',
            $this->securityRoot . '/external_security_files.sha256.sig'
        );
        $externalEntries = $this->parseManifest($externalManifest);
        $this->assertRequiredEntries($externalEntries, [
            'ExternalSecurityVerifier.php',
            'preflight.php',
            'monitor.php',
            'verify.php',
            'scan-known-backdoors.php',
        ]);
        $this->verifyManifestFiles($externalEntries, $this->securityRoot);

        $manifest = $this->verifiedPayload(
            $this->securityRoot . '/deposit_critical_files.sha256',
            $this->securityRoot . '/deposit_critical_files.sha256.sig'
        );

        $allowlist = $this->verifiedPayload(
            $this->securityRoot . '/deposit_address_allowlist.sha256',
            $this->securityRoot . '/deposit_address_allowlist.sha256.sig'
        );

        $entries = $this->parseManifest($manifest);
        $this->assertRequiredEntries($entries, $this->requiredPublicEntries());
        $this->verifyManifestFiles($entries, $this->publicRoot);

        $allowedCount = $this->validateAllowlist($allowlist);

        return [
            'files' => count($entries),
            'external_files' => count($externalEntries),
            'approved_wallet_hashes' => $allowedCount,
        ];
    }

    private function verifiedPayload(string $payloadPath, string $signaturePath): string
    {
        if (!function_exists('sodium_crypto_sign_verify_detached')) {
            throw new RuntimeException('PHP Sodium extension is unavailable.');
        }

        $payload = $this->readFile($payloadPath, self::MAX_PAYLOAD_BYTES, 'payload');
        $signature = base64_decode(trim($this->readFile($signaturePath, 4096, 'signature')), true);
        $publicKey = base64_decode(trim($this->readFile(
            $this->securityRoot . '/deposit_security_public.key',
            4096,
            'public key'
        )), true);

        if (!is_string($publicKey) || strlen($publicKey) !== 32) {
            throw new RuntimeException('Invalid public key.');
        }

        if (!hash_equals(self::PUBLIC_KEY_SHA256, hash('sha256', $publicKey))) {
            throw new RuntimeException('Public key fingerprint mismatch.');
        }

        if (!is_string($signature)
            || strlen($signature) !== 64
            || !sodium_crypto_sign_verify_detached($signature, $payload, $publicKey)) {
            throw new RuntimeException('Detached signature verification failed.');
        }

        return $payload;
    }

    private function parseManifest(string $payload): array
    {
        $entries = [];

        foreach (preg_split('/\R/', $payload) ?: [] as $lineNumber => $line) {
            $line = trim((string) $line);
            if ($line === '' || str_starts_with($line, '#')) {
                continue;
            }

            if (!preg_match('/^([a-f0-9]{64}) {2}([A-Za-z0-9._\/-]+)$/', $line, $match)) {
                throw new RuntimeException('Invalid manifest line ' . ($lineNumber + 1) . '.');
            }

            $relative = $this->safePath($match[2]);
            if (isset($entries[$relative])) {
                throw new RuntimeException('Duplicate manifest entry: ' . $relative);
            }

            $entries[$relative] = strtolower($match[1]);
        }

        if (!$entries) {
            throw new RuntimeException('Signed integrity manifest is empty.');
        }

        return $entries;
    }

    private function requiredPublicEntries(): array
    {
        return [
            '.htaccess',
            'index.php',
            'assets/.htaccess',
            'assets/support/.htaccess',
            'assets/verify/.htaccess',
            'assets/game_aviator/.htaccess',
            'assets/games/.htaccess',
            'assets/images/discover/.htaccess',
            'assets/images/user/profile/.htaccess',
            'assets/images/verify/deposit/.htaccess',
            'assets/images/verify/withdraw/.htaccess',
            'core/app/Http/Controllers/Api/PaymentController.php',
            'core/app/Http/Controllers/Gateway/PaymentController.php',
            'core/app/Http/Middleware/VerifyDepositSecurity.php',
            'core/app/Services/SecureDepositAddressService.php',
            'core/app/Support/DepositSecurityGuard.php',
            'core/app/Support/SignedSecurityFileVerifier.php',
            'core/bootstrap/app.php',
            'core/config/deposit_security.php',
            'core/resources/views/templates/invester/user/payment/manual.blade.php',
            'core/routes/api.php',
            'core/routes/user.php',
            'core/routes/web.php',
        ];
    }

    private function assertRequiredEntries(array $entries, array $required): void
    {
        foreach ($required as $path) {
            if (!isset($entries[$path])) {
                throw new RuntimeException('Required manifest entry is missing: ' . $path);
            }
        }
    }

    private function verifyManifestFiles(array $entries, string $configuredRoot): void
    {
        $root = realpath($configuredRoot);
        if (!is_string($root) || $root === '' || $root === DIRECTORY_SEPARATOR) {
            throw new RuntimeException('Invalid public_html path.');
        }

        $prefix = rtrim(str_replace('\\', '/', $root), '/') . '/';

        foreach ($entries as $relative => $expectedHash) {
            $candidate = $root . DIRECTORY_SEPARATOR . str_replace('/', DIRECTORY_SEPARATOR, $relative);
            $resolved = realpath($candidate);
            $normalized = is_string($resolved) ? str_replace('\\', '/', $resolved) : '';

            if (is_link($candidate)
                || !is_string($resolved)
                || !is_file($resolved)
                || !is_readable($resolved)
                || !str_starts_with($normalized, $prefix)) {
                throw new RuntimeException('Protected file is missing or unsafe: ' . $relative);
            }

            $actualHash = hash_file('sha256', $resolved);
            if (!is_string($actualHash) || !hash_equals($expectedHash, strtolower($actualHash))) {
                throw new RuntimeException('Protected file hash mismatch: ' . $relative);
            }
        }
    }

    private function validateAllowlist(string $payload): int
    {
        $hashes = [];

        foreach (preg_split('/\R/', $payload) ?: [] as $lineNumber => $line) {
            $line = trim((string) $line);
            if ($line === '' || str_starts_with($line, '#')) {
                continue;
            }

            if (!preg_match('/^([a-f0-9]{64})(?: {2}# [A-Za-z0-9 ._:\/-]{1,120})?$/i', $line, $match)) {
                throw new RuntimeException('Invalid signed allowlist line ' . ($lineNumber + 1) . '.');
            }

            $hashes[strtolower($match[1])] = true;
        }

        if (!$hashes) {
            throw new RuntimeException('Signed wallet allowlist is empty.');
        }

        return count($hashes);
    }

    private function safePath(string $path): string
    {
        $path = str_replace('\\', '/', trim($path));
        $segments = explode('/', $path);

        if ($path === ''
            || str_starts_with($path, '/')
            || preg_match('/[\x00-\x1F\x7F]/', $path)
            || in_array('..', $segments, true)
            || in_array('', $segments, true)) {
            throw new RuntimeException('Unsafe manifest path.');
        }

        return $path;
    }

    private function readFile(string $path, int $maximumBytes, string $label): string
    {
        if ($path === '' || is_link($path) || !is_file($path) || !is_readable($path)) {
            throw new RuntimeException('Missing or unsafe security ' . $label . '.');
        }

        $size = filesize($path);
        if (!is_int($size) || $size < 1 || $size > $maximumBytes) {
            throw new RuntimeException('Invalid security ' . $label . ' size.');
        }

        $contents = file_get_contents($path);
        if (!is_string($contents) || $contents === '') {
            throw new RuntimeException('Unable to read security ' . $label . '.');
        }

        return $contents;
    }
}
