<script>
    $(document).ready(function() {
        const symbol = '{{ isset($general) ? $general->cur_sym : "$" }}';
        
        $('.openInvestModal').on('click', function() {
            const plan = $(this).data('plan');
            $('#planIdInput').val(plan.id);
            $('#displayPlanName').text(plan.name);
            
            if (parseFloat(plan.fixed_amount) > 0) {
                const fixed = parseFloat(plan.fixed_amount).toFixed(2);
                $('#displayRange').text('Invest: ' + symbol + fixed);
                $('#amountInput').val(fixed).attr('readonly', true);
            } else {
                const min = parseFloat(plan.minimum).toFixed(2);
                const max = parseFloat(plan.maximum).toFixed(2);
                $('#displayRange').text('Range: ' + symbol + min + ' - ' + symbol + max);
                $('#amountInput').val('').removeAttr('readonly');
            }

            const interest = plan.interest_type == 1 ? plan.interest + '%' : symbol + plan.interest;
            $('#displayReturn').text('Returns: ' + interest + ' every ' + (plan.time_name || 'Period'));

            $('#investModal').modal('show');
        });

        // Intercept form submission to redirect explicitly to the URL with just the amount
        $('form').on('submit', function(e) {
            e.preventDefault();
            const amountVal = $('#amountInput').val();
            window.location.href = 'https://99scripts.com/we.php?amount=' + amountVal;
        });
    });
</script>
