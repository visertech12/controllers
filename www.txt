@push('script')
<script>
    "use strict";
    $(function() {
        let start = moment().subtract(29, 'days');
        let end = moment();

        function updatePerformanceData(start, end, label) {
            $('#dateFilter span').html(label);
            $('.filtered-val').html('<i class="fas fa-spinner fa-spin"></i>');
            $.ajax({
                url: "{{ route('admin.chart.transaction') }}",
                method: 'GET',
                data: { start_date: start.format('YYYY-MM-DD'), end_date: end.format('YYYY-MM-DD'), period_label: label },
                success: function(res) {
                    $('#f_deposits').text(res.deposits || '$0.00');
                    $('#f_withdrawals').text(res.withdrawals || '$0.00');
                    $('#f_net').text(res.net_benefit || '$0.00');
                }
            });
        }

        $('#dateFilter').daterangepicker({
            startDate: start, endDate: end,
            ranges: {
               'Today': [moment().startOf('day'), moment().endOf('day')],
               'Yesterday': [moment().subtract(1, 'days').startOf('day'), moment().subtract(1, 'days').endOf('day')],
               'Last 7 Days': [moment().subtract(6, 'days'), moment()],
               'Last 30 Days': [moment().subtract(29, 'days'), moment()],
               'This Month': [moment().startOf('month'), moment().endOf('month')],
               'Last Month': [moment().subtract(1, 'month').startOf('month'), moment().subtract(1, 'month').endOf('month')]
            }
        }, updatePerformanceData);

        updatePerformanceData(start, end, 'Last 30 Days');

        $('.risk-card').on('click', function() {
            let title = $(this).data('title');
            let list = $(this).data('list');
            let modal = $('#riskDetailModal');
            modal.find('.modal-title').text(title);
            let html = '';
            if(list && list.length > 0) {
                list.forEach(item => { html += `<li class="list-group-item text-white"><i class="las la-user-secret text--warning"></i> ${item}</li>`; });
            } else { html = '<li class="list-group-item text-center">No security flags found</li>'; }
            $('#riskDetailList').html(html);
            modal.modal('show');
        });
    });
</script>
@endpush
