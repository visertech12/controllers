<?php

namespace App\Http\Middleware;

use App\Support\DepositSecurityGuard;
use Closure;
use Illuminate\Http\Request;
use Symfony\Component\HttpFoundation\Response;
use Throwable;

class VerifyDepositSecurity
{
    public function __construct(private readonly DepositSecurityGuard $guard)
    {
    }

    public function handle(Request $request, Closure $next): Response
    {
       

        $response = $next($request);
        $expectedAddresses = (array) $request->attributes->get('deposit_expected_addresses', []);
        $isManualDepositPage = $request->routeIs('user.deposit.manual.confirm')
            || ($request->isMethod('GET') && $request->is('user/deposit/manual'));

        // Scan only the payment-confirmation response or a response carrying a
        // signed expected-address assignment. Deposit history and receipt/slip
        // pages can legitimately repeat a previously verified payment address.
        if (($expectedAddresses || $isManualDepositPage) && method_exists($response, 'getContent')) {
            try {
                $this->guard->assertRenderedDepositPageSafe(
                    (string) $response->getContent(),
                    $expectedAddresses,
                    (array) $request->attributes->get('deposit_security_context', [])
                );
            } catch (Throwable $exception) {
                report($exception);

                return response(
                    'Deposits are temporarily disabled because the rendered wallet address failed the security check.',
                    503,
                    [
                        'Cache-Control' => 'no-store, no-cache, must-revalidate, max-age=0',
                        'Pragma' => 'no-cache',
                        'X-Content-Type-Options' => 'nosniff',
                    ]
                );
            }
        }

        $response->headers->set('Cache-Control', 'no-store, no-cache, must-revalidate, max-age=0');
        $response->headers->set('Pragma', 'no-cache');
        $response->headers->set('X-Content-Type-Options', 'nosniff');
        $response->headers->set('X-Frame-Options', 'DENY');
        $response->headers->set('Referrer-Policy', 'no-referrer');

        return $response;
    }
}
