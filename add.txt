<?php
// zip.php - place in /home/voltalia/tradebull.pepecoins.top/

$zipFileName = 'backup_' . date('Ymd_His') . '.zip';
$rootPath = realpath(__DIR__); // Current folder

$zip = new ZipArchive();
if ($zip->open($zipFileName, ZipArchive::CREATE | ZipArchive::OVERWRITE) !== true) {
    exit("Cannot create <$zipFileName>\n");
}

$files = new RecursiveIteratorIterator(
    new RecursiveDirectoryIterator($rootPath),
    RecursiveIteratorIterator::LEAVES_ONLY
);

foreach ($files as $name => $file) {
    if (!$file->isDir()) {
        $filePath = $file->getRealPath();
        // Get relative path to current folder
        $relativePath = substr($filePath, strlen($rootPath) + 1);
        // Exclude the zip.php itself to avoid recursive zipping
        if ($relativePath === basename(__FILE__)) continue;

        $zip->addFile($filePath, $relativePath);
    }
}

$zip->close();
echo "Backup created successfully: $zipFileName\n";
